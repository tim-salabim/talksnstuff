flist <- lapply(seq(files), function(i) {
    tmp <- read.csv(files[i])
    tmp$p_RT_NRT <- tmp$p_RT_NRT * 25.4  # precip data stored in inches
    tmp$name <- rep(names[i], nrow(tmp))
    tmp$moda <- sprintf("%04.0f", tmp$moda)
    tmp$month <- as.POSIXct(strptime(paste(tmp$year, substr(tmp$moda, 1, 2), 
        "01", sep = ""), format = "%Y%m%d"))
    tmp$date <- as.POSIXct(strptime(paste(tmp$year, tmp$moda, sep = ""), format = "%Y%m%d"))
    tmp$p_RT_NRT_mean <- rep(mean(tmp$p_RT_NRT, na.rm = TRUE), nrow(tmp))
    tmp$p_RT_NRT_dev <- tmp$p_RT_NRT - tmp$p_RT_NRT_mean

    start_time <- paste(tmp$year[1], "01-01", sep = "-")
    end_time <- paste(tmp$year[nrow(tmp)], "12-31", sep = "-")
    time_step <- 60 * 60 * 24
    date_from <- as.POSIXct(start_time)
    date_to <- as.POSIXct(end_time)
    deltat <- time_step

    ## create regular time series
    tseries <- data.frame(date = seq(from = date_from, to = date_to, by = deltat))
    tmp2 <- merge(tseries, tmp, all = TRUE)
    tmp3 <- merge(tmp2, types, all = FALSE)
    return(tmp3)
})

findMaxCCF<- function(a, b, ...)
{
 d <- ccf(a, b, plot = FALSE, ...)
 cor = d$acf[,,1]
 lag = d$lag[,,1]
 res = data.frame(cor,lag)
 res_max = res[which.max(res$cor),]
 return(res_max)
}

findMaxIOA <- function(a, b, lags = c(-20:20)) {
  tmp <- lapply(seq(lags), function(i) {
    lag <- lags[i]
     IOA(a, Lag(b, shift = lag))
      })
  ind <- which(unlist(tmp) == max(unlist(tmp)))
  return(c(tmp[[ind]], lags[ind]))
  }
